cmake_minimum_required(VERSION 3.25)

# Minimal iOS build for Phase 0 - CPU emulator only

set(PLATFORM_IOS ON)

project(Cemu_iOS VERSION 2.0.0)

list(APPEND CMAKE_MODULE_PATH "${PROJECT_SOURCE_DIR}/cmake")

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Detect iOS build
if(CMAKE_SYSTEM_NAME STREQUAL "iOS")
    set(PLATFORM_IOS ON)
    message(STATUS "Building Cemu for iOS")
endif()

add_compile_definitions(EMULATOR_HASH="phase0")
add_compile_definitions(EMULATOR_VERSION_MAJOR=0)
add_compile_definitions(EMULATOR_VERSION_MINOR=0)
add_compile_definitions(EMULATOR_VERSION_PATCH=0)

set_property(GLOBAL PROPERTY USE_FOLDERS ON)

find_package(Threads REQUIRED)

# Minimal dependencies for CPU emulator only
set(CMAKE_PREFIX_PATH "/usr/local/opt/libzip;/usr/local/opt/zlib;/usr/local/opt/openssl@3" ${CMAKE_PREFIX_PATH})

# Core emulation without external dependencies
add_subdirectory(src/Common)

# Only compile the Cafe core (Wii U emulation)
add_subdirectory(src/Cafe)

# Minimal config system
add_subdirectory(src/config)

# Input handling
add_subdirectory(src/input)

# iOS-specific UI layer
add_subdirectory(src/gui)

# Main emulator core
add_library(CemuCore STATIC
    src/main.cpp
    src/mainLLE.cpp
)

target_link_libraries(CemuCore PRIVATE
    CemuCommon
    CemuCafe
    CemuConfig
    CemuInput
    CemuGui
    Threads::Threads
)

# iOS frameworks
target_link_libraries(CemuCore PRIVATE
    "-framework Foundation"
    "-framework Metal"
    "-framework MetalKit"
    "-framework UIKit"
)

message(STATUS "iOS Phase 0 build configured successfully")
message(STATUS "  Target: CemuCore (static library)")
message(STATUS "  Platform: ${CMAKE_SYSTEM_PROCESSOR}")
