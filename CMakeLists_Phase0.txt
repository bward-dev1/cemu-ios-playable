cmake_minimum_required(VERSION 3.25)

# Cemu iOS - Phase 0: CPU Emulator Core Only
# Minimal build with zero external dependencies

project(Cemu_iOS_Phase0 VERSION 2.0.0)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add only the absolute minimum
add_subdirectory(src/Common)
add_subdirectory(src/Cafe)
add_subdirectory(src/util)

# Create iOS static library with just the emulator
add_library(CemuCore STATIC src/main.cpp)

target_link_libraries(CemuCore PRIVATE
    CemuCommon
    CemuCafe
)

# iOS frameworks
target_link_libraries(CemuCore PRIVATE
    "-framework Foundation"
    "-framework Metal"
    "-framework MetalKit"
    "-framework UIKit"
)

message(STATUS "✓ Phase 0 iOS Build: Minimal CPU emulator only")
